```c++
#include<iostream>
using namespace std;

int main()
{
    int n, a;
    cin >> n;
    for(int i = 0;i < n;i++)
    {

        cin >> a;
        int count = 0;
        while(a)
        {
            a = a & (a - 1); // 每次进行这个操作二进制数中最右边的1都会变成0，可以统计有多少个1
            count++;
        }
        cout << count << " ";
    }
    return 0;
}
```
